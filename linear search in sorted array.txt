#include<stdio.h>
int findElement(int arr[], int n, int key);
 
 int main()
 {
     int arr[50] = {1, 3, 5, 6, 8, 10};
     int position, n ;
     n= sizeof(arr)/sizeof(arr[0]);
     int key = 10;
     position  = findElement(arr, n, key);
     if (position == -1)
     {
         printf ("\n Element not found");
     }
     else 
     {
         printf ("\n %d is found at position :%d", key, position+1);
     }
     return 0;
     
 }
 
 
int findElement(int arr[], int n, int key)
 {
     int i;
     for( i=0; i<n; i++)
        if (arr[i] == key)
        return i;
        
        return -1;
 }
