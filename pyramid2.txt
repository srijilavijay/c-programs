Q. C program to print following pattern

1
12
123
1234
12345


#include<stdio.h>
int main()
{
    int i,j,n;
    printf("Enter the limit:");
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        printf("\n");
        for(j=1;j<=i; j++)
        {
            printf("%d", j);
        }
    }
    return 0;
}
